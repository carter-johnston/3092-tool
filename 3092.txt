	private static String generateRandomDateOfBirth(Random random) {
		int year = random.nextInt(50) + 1950; // Generate year between 1950 and 1999
		int month = random.nextInt(12) + 1; // Generate month between 1 and 12
		int day = random.nextInt(28) + 1; // Generate day between 1 and 28 (for simplicity, not accounting for different
											// days in each month)
		return String.format("%04d%02d%02d", year, month, day );
	}

	private static String generateRandomSocialSecurityNumber(Random random) {
		int group1 = random.nextInt(899) + 100; // Generate 3-digit number between 100 and 999
		int group2 = random.nextInt(99); // Generate 2-digit number between 0 and 99
		int group3 = random.nextInt(9999); // Generate 4-digit number between 0 and 9999
		return String.format("%03d%02d%04d", group1, group2, group3);
	}

	private static String generateRandomEmailAddress(String firstName, String lastName) {
		String[] domains = { "usbank.com" };
		Random random = new Random();
		String domain = domains[random.nextInt(domains.length)];
		return firstName.toLowerCase() + "." + lastName.toLowerCase() + "@" + domain;
	}

	private static String generateRandomPhoneNumber(Random random) {
		StringBuilder phoneNumber = new StringBuilder();
		for (int i = 0; i < 10; i++) {
			phoneNumber.append(random.nextInt(10));
		}
		return phoneNumber.toString();
	}
